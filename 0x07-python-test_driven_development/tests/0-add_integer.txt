# 0-add_integer.py


def add_integer(a, b=98):
    """Returns a plus b.

    Works with integers:

    >>> add_integer(5, 4)
    9

    >>> add_integer(-5, 4)
    -1

    >>> add_integer(-5, -4)
    -9

    Does not work with non-ints:

    >>> add_integer('a', 1)
    a must be an integer

    >>> add_integer(3, 'n')
    b must be an integer
    """
    return int(a) + int(b)
